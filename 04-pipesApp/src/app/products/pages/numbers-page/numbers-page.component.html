Basics Angular Pipes

Basics Angular Pipes, included in CommonModule.

Consider having the following publics properties in your component: 

        
            public
             amount
            : number = 
            123456.789
        
        
    
{{ "{{ amount | number }}" }} produces {{ amount | number }}
{{ "{{ amount | number: '1.0-2' }}" }} produces {{ amount | number: '1.0-2' }}
 
{{ "{{ amount | currency: 'NIO' }}" }} produces {{ amount | currency: 'NIO' }}
{{ "{{ amount | currency: 'EUR': 'symbol-narrow' }}" }} produces {{ amount | currency: 'EUR': 'symbol-narrow' }}
{{ "{{ amount | currency: 'USD': 'symbol-narrow': '1.4-4' }}" }} produces {{ amount | currency: 'USD': 'symbol-narrow': '1.4-4' }}
 
{{ "{{ percent | percent }}" }} produces {{ percent | percent }}
{{ "{{ percent | percent: '1.0-2' }}" }} produces {{ percent | percent: '1.0-2' }}
{{ "{{ percent | percent: '1.4-4' }}" }} produces {{ percent | percent: '1.4-4' }}
